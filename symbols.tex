\NewDocumentCommand{\DEFradius}{O{1}}{%
	\ifnum #1>0 %
		\frac{r^{#1}}{R^{#1}} %
	\else %
		\ifnum #1=-1 %
			\frac{R}{r} %
		\else %
			\frac{R^{\StrGobbleLeft{#1}{1}}}{r^{\StrGobbleLeft{#1}{1}}} %
		\fi %
	\fi %
}
